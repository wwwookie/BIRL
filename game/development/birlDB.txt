CREATE TABLE type(

	id				SERIAL		PRIMARY KEY,
	name			VARCHAR(50)	NOT NULL,
	good_against	VARCHAR(50)	NOT NULL,
	bad_against		VARCHAR(50)	NOT NULL
);

CREATE TABLE character(

	id			SERIAL		PRIMARY KEY,
	name		VARCHAR(50)	UNIQUE NOT NULL,
	lvl_scaling	DOUBLE		NOT NULL,
	type		VARCHAR(50)	NOT NULL	references type(name),
	ability		VARCHAR(500)
);

CREATE TABLE fighter(
	
	id		SERIAL		PRIMARY KEY references character(id),
	lvl		SERIAL		NOT NULL,
	hp		INT			NOT NULL,
	move1	SERIAL		references move(id),
	move2	SERIAL		references move(id),
	move3	SERIAL		references move(id),
	gear	SERIAL		references gear(id),
	team	SERIAL		references team(id),

	CONSTRAINT valid_number
		CHECK(LVL <= 100)
);

CREATE TABLE moves(

	id			SERIAL		PRIMARY KEY,
	name		VARCHAR(50)	UNIQUE,
	type		VARCHAR(50)	references type(name),
	damage		INT,
	heal		INT,
	effect		SERIAL		references effect(id)
);

CREATE TABLE gear(

	id		SERIAL			PRIMARY KEY,
	name		VARCHAR(50)	UNIQUE NOT NULL,
	damage		INT			NOT NULL,
	heal		INT			NOT NULL,
	effect		SERIAL		references effect(id)
);

CREATE TABLE effect(

	id		SERIAL		PRIMARY KEY,
	text	VARCHAR(500)	NOT NULL,
);

CREATE TABLE person(

	id			SERIAL		PRIMARY KEY,
	name		VARCHAR(50)	UNIQUE	NOT NULL,
	password	VARCHAR(50)	NOT NULL,
	victory_points	INT,
	team_id		SERIAL		references team(id),
	user_lvl	SERIAL,
	CONSTRAINT valid_number
		CHECK(role <= 2)
);

CREATE TABLE team(

	id			SERIAL		PRIMARY KEY,
	owner		SERIAL		references person(id),
	fighter1	SERIAL		references fighter(id),
	fighter2	SERIAL		references fighter(id),
	fighter3	SERIAL		references fighter(id),
	fighter4	SERIAL		references fighter(id),
	fighter5	SERIAL		references fighter(id),
);
